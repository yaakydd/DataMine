### **Day 1-3: The Very Basics**

#### 1. **Variables & Data Types**
```javascript
// Variables - Containers for data
let name = "John";           // String (text)
const age = 25;              // Number
let isStudent = true;        // Boolean (true/false)
let nothing = null;          // Null (intentionally empty)
let notDefined;              // Undefined (not assigned yet)

// Arrays - Lists of items
let fruits = ["apple", "banana", "orange"];
console.log(fruits[0]);      // "apple" (arrays start at 0)

// Objects - Collections of properties
let person = {
    name: "John",
    age: 25,
    city: "Kumasi"
};
console.log(person.name);    // "John"
```

**ğŸ’¡ Plain English:**
- **Variables** are like labeled boxes that hold information
- **let** = can change later, **const** = stays the same forever
- **Arrays** are like shopping lists (ordered)
- **Objects** are like contact cards (key-value pairs)

---

#### 2. **Operators & Expressions**
```javascript
// Arithmetic
let sum = 10 + 5;           // 15
let product = 10 * 5;       // 50
let remainder = 10 % 3;     // 1 (modulo - remainder after division)

// Comparison
10 > 5                      // true
10 === 10                   // true (strict equal - checks value AND type)
10 == "10"                  // true (loose equal - only checks value)
10 === "10"                 // false (different types)

// Logical
true && false               // false (AND - both must be true)
true || false               // true (OR - at least one must be true)
!true                       // false (NOT - flips the value)

// String concatenation
let greeting = "Hello" + " " + "World";  // "Hello World"
let template = `Hello ${name}`;          // Template literal (modern way)
```

---

#### 3. **Conditionals (If/Else)**
```javascript
let age = 20;

if (age >= 18) {
    console.log("You're an adult");
} else if (age >= 13) {
    console.log("You're a teenager");
} else {
    console.log("You're a child");
}

// Ternary operator (shorthand if/else)
let status = age >= 18 ? "Adult" : "Minor";

// Switch statement
let day = "Monday";
switch(day) {
    case "Monday":
        console.log("Start of week");
        break;
    case "Friday":
        console.log("Almost weekend!");
        break;
    default:
        console.log("Regular day");
}
```

**ğŸ’¡ Plain English:**
- **if/else** = "If this is true, do this. Otherwise, do that."
- **Ternary** = Shortcut for simple if/else
- **Switch** = Multiple if/else checks in a cleaner way

### **Day 4-7: Functions & Loops**

#### 4. **Functions (Reusable Code Blocks)**
```javascript
// Function Declaration
function greet(name) {
    return `Hello, ${name}!`;
}
console.log(greet("Yaa"));  // "Hello, Yaa!"

// Arrow Function (modern, shorter syntax)
const greet = (name) => `Hello, ${name}!`;

// Function with multiple parameters
const add = (a, b) => a + b;
console.log(add(5, 3));     // 8

// Function with default parameter
const greet = (name = "Guest") => `Hello, ${name}!`;
console.log(greet());       // "Hello, Guest!"

// Callback functions (function as parameter)
function processData(data, callback) {
    let result = data * 2;
    callback(result);
}

processData(5, (result) => {
    console.log(result);    // 10
});
```

**ğŸ’¡ Plain English:**
- **Functions** are like recipes - you define once, use many times
- **Parameters** are ingredients you pass in
- **Return** gives you the final dish
- **Arrow functions** are just shorter syntax for functions

---

#### 5. **Loops (Repeating Actions)**
```javascript
// For loop - when you know how many times
for (let i = 0; i < 5; i++) {
    console.log(i);  // 0, 1, 2, 3, 4
}

// While loop - when you don't know how many times
let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

// For...of loop - looping through arrays (modern)
let fruits = ["apple", "banana", "orange"];
for (let fruit of fruits) {
    console.log(fruit);
}

// forEach - array method (most common)
fruits.forEach((fruit, index) => {
    console.log(`${index}: ${fruit}`);
});
```

**ğŸ’¡ Plain English:**
- **for** = "Do this 5 times"
- **while** = "Keep doing this until something changes"
- **for...of** = "Do this for each item in the list"
- **forEach** = Modern way to loop through arrays

## ğŸ”¥ Phase 2: Core Concepts (Week 3-4)

### **Array Methods (Super Important!)**
```javascript
let numbers = [1, 2, 3, 4, 5];

// map - Transform each item
let doubled = numbers.map(num => num * 2);
// [2, 4, 6, 8, 10]

// filter - Keep only items that match
let evenNumbers = numbers.filter(num => num % 2 === 0);
// [2, 4]

// find - Get first item that matches
let firstEven = numbers.find(num => num % 2 === 0);
// 2

// reduce - Combine all items into one value
let sum = numbers.reduce((total, num) => total + num, 0);
// 15

// some - Check if ANY item matches
let hasEven = numbers.some(num => num % 2 === 0);
// true

// every - Check if ALL items match
let allPositive = numbers.every(num => num > 0);
// true

// includes - Check if array contains value
numbers.includes(3);  // true

// sort - Arrange items
numbers.sort((a, b) => a - b);  // Ascending
numbers.sort((a, b) => b - a);  // Descending
```

**ğŸ’¡ Plain English:**
These are your POWER TOOLS. You'll use them constantly in React:
- **map** = Transform a list (like converting Celsius to Fahrenheit)
- **filter** = Keep only what you want (like filtering search results)
- **find** = Find one specific item
- **reduce** = Combine everything into one value (like totaling prices)

---

### **Object Methods**
```javascript
let person = {
    name: "John",
    age: 25,
    city: "Kumasi"
};

// Get all keys
Object.keys(person);        // ["name", "age", "city"]

// Get all values
Object.values(person);      // ["John", 25, "Kumasi"]

// Get key-value pairs
Object.entries(person);     
// [["name", "John"], ["age", 25], ["city", "Kumasi"]]

// Copy object (shallow copy)
let copy = { ...person };

// Merge objects
let extended = { ...person, job: "Developer" };

// Destructuring - extract values
let { name, age } = person;
console.log(name);          // "John"
```

---

### **Spread & Rest Operators**
```javascript
// Spread (...) - Expand/unpack items
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];
let combined = [...arr1, ...arr2];  // [1, 2, 3, 4, 5, 6]

let obj1 = { name: "John" };
let obj2 = { age: 25 };
let merged = { ...obj1, ...obj2 };  // { name: "John", age: 25 }

// Rest (...) - Collect remaining items
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}
sum(1, 2, 3, 4);  // 10
```

**ğŸ’¡ Plain English:**
- **Spread (...)** = "Unpack this"
- **Rest (...)** = "Pack the rest into an array"

### **Event Handling**
```javascript
// Click event
button.addEventListener('click', () => {
    console.log('Button clicked!');
});

// Form submit
form.addEventListener('submit', (e) => {
    e.preventDefault();  // Stop page reload
    let input = document.querySelector('input').value;
    console.log(input);
});

// Input change
input.addEventListener('input', (e) => {
    console.log(e.target.value);
});

// Mouse events
element.addEventListener('mouseenter', () => {});
element.addEventListener('mouseleave', () => {});

// Keyboard events
document.addEventListener('keydown', (e) => {
    console.log(e.key);  // Which key was pressed
});
```

**ğŸ’¡ Plain English:**
- **addEventListener** = "When this happens, do that"
- **e.preventDefault()** = "Don't do the default action"
- **e.target** = The element that triggered the event

## âš¡ Phase 4: Async JavaScript (Week 6)

### **Promises**
```javascript
// Creating a promise
let promise = new Promise((resolve, reject) => {
    let success = true;
    
    if (success) {
        resolve("Success!");
    } else {
        reject("Error!");
    }

    .then(result => console.log(result))
```

---

### **Async/Await (Modern Way)**
```javascript
// Fetch data from API
async function fetchData() {
    try {
        let response = await fetch('https://api.example.com/data');
        let data = await response.json();
        console.log(data);
    } catch (error) {
        console.log('Error:', error);
    }
}

// More examples
async function getData() {
    // Wait for multiple requests
    let [users, posts] = await Promise.all([
        fetch('/api/users').then(r => r.json()),
        fetch('/api/posts').then(r => r.json())
    ]);
    
    return { users, posts };
}
```

**ğŸ’¡ Plain English:**
- **Promises** = "I'll get you that data... eventually"
- **async/await** = Makes async code look like normal code
- **try/catch** = Handle errors gracefully

### **Destructuring**
```javascript
// Array destructuring
let [first, second, ...rest] = [1, 2, 3, 4, 5];
// first = 1, second = 2, rest = [3, 4, 5]

// Object destructuring
let person = { name: "John", age: 25, city: "Kumasi" };
```

---

### **Optional Chaining & Nullish Coalescing**
```javascript
// Optional chaining (?.)
let user = {
    profile: {
        name: "John"
    }
};

// Old way
let name = user && user.profile && user.profile.name;

// New way
let name = user?.profile?.name;

// Nullish coalescing (??)
let value = null;
let result = value ?? "default";  // "default"

// vs OR operator
let result = 0 || "default";      // "default" (0 is falsy)
let result = 0 ?? "default";      // 0 (only null/undefined trigger ??)
```

### **React Basics**
```javascript
import React, { useState, useEffect } from 'react';

// Functional Component
function Counter() {
    // State - data that changes
    const [count, setCount] = useState(0);
    
    // Effect - runs after render
    useEffect(() => {
        document.title = `Count: ${count}`;
    }, [count]);  // Run when count changes
    
    return (
        <div>
            <p>Count: {count}</p>
            <button onClick={() => setCount(count + 1)}>
                Increment
            </button>
        </div>
    );
}

export default Counter;
```

## ğŸ’¡ Key Concepts You MUST Master

### **1. Variables & Scope**
```javascript
// Global scope
let globalVar = "I'm global";

function myFunction() {
    // Function scope
    let localVar = "I'm local";
    
    if (true) {
        // Block scope
        let blockVar = "I'm in a block";
    }
}
```

### **2. This Keyword**
```javascript
const person = {
    name: "John",
    greet: function() {
        console.log(`Hello, ${this.name}`);
    },
    greetArrow: () => {
        console.log(`Hello, ${this.name}`);  // undefined!
    }
};

// Regular function: 'this' refers to person
person.greet();  // "Hello, John"

// Arrow function: 'this' refers to parent scope
person.greetArrow();  // "Hello, undefined"
```

### **3. Closures**
```javascript
function createCounter() {
    let count = 0;
    
    return {
        increment: () => ++count,
        decrement: () => --count,
        getCount: () => count
    };
}

const counter = createCounter();
counter.increment();  // 1
counter.increment();  // 2
counter.getCount();   // 2
```
